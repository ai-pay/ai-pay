## Example
<ImageGenerationExample>
```typescript
import { imageGeneration } from "ai-pay/apis"
import { ImageGenerationRequest } from "ai-pay/models";

async function callImageGeneration() {  
  const request: ImageGenerationRequest = INPUT;

  const {
    error,
    debugError,
    data,
  } = await imageGeneration(request);

  if (error) {
    displayErrorToUser(request, error);
    console.error("AI Pay debug error", debugError)
    return;
  } 

  const { 
    imageUrls 
  } = data;
  prependImagesToUi(request, imageUrls);
}
```
</ImageGenerationExample>
